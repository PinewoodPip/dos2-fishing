// Apply Harried when proccing Seasick.
// Hook into this to perform effects when Seasick detonates.
PROC PROC_PIP_Seasick_Detonate((CHARACTERGUID)_Target, (CHARACTERGUID)_StatusSource, (CHARACTERGUID)_Attacker)
THEN
PROC_AMER_BatteredHarried_DirectManip(_Target, 1, 1, _Attacker);

// Return to Crab SI3: remove Spellcaster Finesse when status expires.
// This hook is necessary as there's no way to bind SpecialLogic to a status.
PROC PROC_AMER_GEN_FilteredStatus_Removed((CHARACTERGUID)_Char, (CHARACTERGUID)_Source, "PIP_Polymorphed_Crab")
AND
DB_PIP_Fishing_HasCrabInfusMageFin(_Char)
THEN
PROC_AMER_UI_Ascension_SpecialLogic_Add(_Char, "", -1.0);
NOT DB_PIP_Fishing_HasCrabInfusMageFin(_Char);

// Prepare Sashimi SI3: if target dies while casting, gain Slashimi for X turns.
PROC PROC_AMER_ExtendedStat_ProcOnDeath((CHARACTERGUID)_Char, (CHARACTERGUID)_Source, "PIP_Source_Infusion_PrepareSashimi", (REAL)_Amount)
AND
RealProduct(_Amount, 6.0, _Duration)
THEN
ApplyStatus(_Source, "PIP_FISHING_SLASHIMI", _Duration, 0, _Source);
PROC_AMER_TempStat_StatusMod_AddExtendedStat(_Source, "PIP_FISHING_SLASHIMI", "PIP_SeasickOnHit", "", "", "", 1.0, "PIP_Fishing_PrepareSashimi_Infus3", 1); // Should be called after applying the status (since we're not using SkillCast timer).

// Apply floating status to crab form.
IF CharacterStatusApplied(_Char, "PIP_Polymorphed_Crab", (CHARACTERGUID)_Source)
AND
GetStatusTurns(_Char, "PIP_Polymorphed_Crab", _Turns)
AND
Real(_Turns, _TurnDuration)
AND
RealProduct(_TurnDuration, 6.0, _Duration)
THEN
ApplyStatus(_Char, "PIP_FISHING_CRAB_FLOATING", _Duration, 1, _Char);

// Clear floating status when losing crab form.
IF CharacterStatusRemoved(_Char, "PIP_Polymorphed_Crab", (CHARACTERGUID)_Source)
THEN
RemoveStatus(_Char, "PIP_FISHING_CRAB_FLOATING");
