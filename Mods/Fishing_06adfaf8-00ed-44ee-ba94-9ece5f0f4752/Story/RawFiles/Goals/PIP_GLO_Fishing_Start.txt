// This is necessary as the ClassType/Equipment fields are not populated after CC;
// they return to the values of the original template of the character (ex. Enchanter for Lohse)

AND
CharacterHasSkill(_Char, "Target_PIP_Fishing_ReelIn", 1) // Should be kept in sync with the skillset. More reliable than checking for the fishing rod template, as more might've been added to the ItemProgressionVisuals.
THEN
SetTag(_Char, "PIP_Fishing_Fishermancer");
