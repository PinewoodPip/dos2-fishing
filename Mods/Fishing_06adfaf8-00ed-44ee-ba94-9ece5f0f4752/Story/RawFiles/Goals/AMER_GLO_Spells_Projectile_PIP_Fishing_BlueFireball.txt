// Replace target randomly when another one is found
PROC PROC_PIP_Source_Infusion_BlueFireBallRandTarget((CHARACTERGUID)_SourceChar, (CHARACTERGUID)_Char)
AND

AND

THEN
NOT DB_PIP_Source_Infusion_BlueFireBallRandomTarget(_SourceChar, _Target);

// Set target if no other was found yet
PROC PROC_PIP_Source_Infusion_BlueFireBallRandTarget((CHARACTERGUID)_SourceChar, (CHARACTERGUID)_Char)
AND
NOT DB_PIP_Source_Infusion_BlueFireBallRandomTarget(_SourceChar, _)
THEN
